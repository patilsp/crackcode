### 1. Difference between `let`, `const`, and `var`

```js
function declarationTest() {
  var a = 1;       // Function-scoped
  let b = 2;       // Block-scoped
  const c = 3;     // Block-scoped + immutable
  
  if (true) {
    var a = 4;     // Same variable (hoisted)
    let b = 5;     // New block-scoped variable
    const c = 6;   // New block-scoped constant
    console.log(a, b, c); // 4, 5, 6
  }
  
  console.log(a, b, c); // 4, 2, 3 (only 'a' leaked)
}
